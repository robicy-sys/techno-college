Enregistrer les scores dans Google Sheets

Pour créer ce script : ouvre Google Sheets → Extensions → Apps Script → colle le code ci-dessous → Déployer → Application web → Tout le monde. 

function doPost(e) {
  const d = JSON.parse(e.postData.contents);
  const s = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
  if (s.getLastRow() === 0)
    s.appendRow(['Date','Prénom','Fiche','Score','Total','%']);
  s.appendRow([new Date(d.date), d.nom, d.titre,
               d.score, d.total,
               Math.round(d.score/d.total*100)+'%']);
  return ContentService.createTextOutput('ok');
}
